package db

import (
	"database/sql"
	"fmt"
	"os"
	"path/filepath"

	_ "github.com/mattn/go-sqlite3"
)

// queryable is the interface satisfied by both *sql.DB and *sql.Tx,
// allowing all DB methods to work in both regular and transactional contexts.
type queryable interface {
	Exec(query string, args ...any) (sql.Result, error)
	Query(query string, args ...any) (*sql.Rows, error)
	QueryRow(query string, args ...any) *sql.Row
}

type DB struct {
	q      queryable // used for all queries — either *sql.DB or *sql.Tx
	sqlDB  *sql.DB   // kept for Close() and BeginTx()
}

func Open(path string) (*DB, error) {
	if err := os.MkdirAll(filepath.Dir(path), 0755); err != nil {
		return nil, fmt.Errorf("create directory: %w", err)
	}

	sqlDB, err := sql.Open("sqlite3", path)
	if err != nil {
		return nil, fmt.Errorf("open database: %w", err)
	}

	pragmas := []string{
		"PRAGMA journal_mode = WAL",
		"PRAGMA busy_timeout = 5000",
		"PRAGMA foreign_keys = ON",
	}
	for _, p := range pragmas {
		if _, err := sqlDB.Exec(p); err != nil {
			sqlDB.Close()
			return nil, fmt.Errorf("set pragma %q: %w", p, err)
		}
	}

	d := &DB{q: sqlDB, sqlDB: sqlDB}
	if err := d.migrate(); err != nil {
		sqlDB.Close()
		return nil, fmt.Errorf("migrate: %w", err)
	}
	return d, nil
}

func (d *DB) Close() error {
	return d.sqlDB.Close()
}

// BeginTx starts a transaction and returns a new DB whose methods
// all execute within that transaction.
func (d *DB) BeginTx() (*DB, error) {
	tx, err := d.sqlDB.Begin()
	if err != nil {
		return nil, err
	}
	return &DB{q: tx, sqlDB: d.sqlDB}, nil
}

// Commit commits the transaction. Only valid on a DB returned by BeginTx.
func (d *DB) Commit() error {
	tx, ok := d.q.(*sql.Tx)
	if !ok {
		return fmt.Errorf("not in a transaction")
	}
	return tx.Commit()
}

// Rollback rolls back the transaction. Only valid on a DB returned by BeginTx.
func (d *DB) Rollback() error {
	tx, ok := d.q.(*sql.Tx)
	if !ok {
		return fmt.Errorf("not in a transaction")
	}
	return tx.Rollback()
}

func (d *DB) migrate() error {
	schema := `
	CREATE TABLE IF NOT EXISTS tasks (
		id          TEXT PRIMARY KEY,
		title       TEXT NOT NULL,
		description TEXT NOT NULL DEFAULT '',
		status      TEXT NOT NULL DEFAULT 'open'
		            CHECK (status IN ('open','active','blocked','done')),
		block_reason TEXT,
		assignee    TEXT,
		parent_id   TEXT REFERENCES tasks(id),
		created_at  TEXT NOT NULL DEFAULT (datetime('now')),
		updated_at  TEXT NOT NULL DEFAULT (datetime('now'))
	);

	CREATE TABLE IF NOT EXISTS deps (
		blocker_id  TEXT NOT NULL REFERENCES tasks(id),
		blocked_id  TEXT NOT NULL REFERENCES tasks(id),
		PRIMARY KEY (blocker_id, blocked_id)
	);

	CREATE TABLE IF NOT EXISTS notes (
		id          INTEGER PRIMARY KEY AUTOINCREMENT,
		task_id     TEXT NOT NULL REFERENCES tasks(id),
		content     TEXT NOT NULL,
		author      TEXT,
		created_at  TEXT NOT NULL DEFAULT (datetime('now'))
	);

	CREATE TRIGGER IF NOT EXISTS tasks_updated_at
	AFTER UPDATE ON tasks
	WHEN NEW.updated_at = OLD.updated_at
	BEGIN
		UPDATE tasks SET updated_at = datetime('now') WHERE id = OLD.id;
	END;
	`
	_, err := d.q.Exec(schema)
	return err
}

package db

import (
	"database/sql"
	"fmt"
	"time"

	"github.com/dispatch-ai/dispatch/internal/id"
)

type Task struct {
	ID          string  `json:"id"`
	Title       string  `json:"title"`
	Description string  `json:"description"`
	Status      string  `json:"status"`
	BlockReason *string `json:"block_reason"`
	Assignee    *string `json:"assignee"`
	ParentID    *string `json:"parent_id"`
	CreatedAt   string  `json:"created_at"`
	UpdatedAt   string  `json:"updated_at"`
}

func (d *DB) AddTask(title, description, parentID, afterID string) (*Task, error) {
	taskID := id.Generate()
	// Check collision
	for {
		var exists bool
		err := d.q.QueryRow("SELECT EXISTS(SELECT 1 FROM tasks WHERE id=?)", taskID).Scan(&exists)
		if err != nil {
			return nil, err
		}
		if !exists {
			break
		}
		taskID = id.Generate()
	}

	var pID *string
	if parentID != "" {
		// Verify parent exists
		if _, err := d.GetTask(parentID); err != nil {
			return nil, fmt.Errorf("parent task %q not found", parentID)
		}
		pID = &parentID
	}

	now := time.Now().UTC().Format("2006-01-02 15:04:05")
	_, err := d.q.Exec(
		`INSERT INTO tasks (id, title, description, parent_id, created_at, updated_at)
		 VALUES (?, ?, ?, ?, ?, ?)`,
		taskID, title, description, pID, now, now,
	)
	if err != nil {
		return nil, fmt.Errorf("insert task: %w", err)
	}

	if afterID != "" {
		if err := d.AddDep(afterID, taskID); err != nil {
			return nil, fmt.Errorf("add dependency: %w", err)
		}
	}

	return d.GetTask(taskID)
}

func (d *DB) GetTask(id string) (*Task, error) {
	t := &Task{}
	err := d.q.QueryRow(
		`SELECT id, title, description, status, block_reason, assignee,
		        parent_id, created_at, updated_at
		 FROM tasks WHERE id = ?`, id,
	).Scan(&t.ID, &t.Title, &t.Description, &t.Status, &t.BlockReason,
		&t.Assignee, &t.ParentID, &t.CreatedAt, &t.UpdatedAt)
	if err == sql.ErrNoRows {
		return nil, fmt.Errorf("task %q not found", id)
	}
	if err != nil {
		return nil, err
	}
	return t, nil
}

func (d *DB) EditTask(id string, title, description *string) (*Task, error) {
	if title == nil && description == nil {
		return d.GetTask(id)
	}
	task, err := d.GetTask(id)
	if err != nil {
		return nil, err
	}
	if title != nil {
		task.Title = *title
	}
	if description != nil {
		task.Description = *description
	}
	_, err = d.q.Exec(
		`UPDATE tasks SET title=?, description=? WHERE id=?`,
		task.Title, task.Description, id,
	)
	if err != nil {
		return nil, fmt.Errorf("update task: %w", err)
	}
	return d.GetTask(id)
}

package db

import "fmt"

func (d *DB) AddDep(blockerID, blockedID string) error {
	// Verify both tasks exist
	if _, err := d.GetTask(blockerID); err != nil {
		return fmt.Errorf("blocker task %q not found", blockerID)
	}
	if _, err := d.GetTask(blockedID); err != nil {
		return fmt.Errorf("blocked task %q not found", blockedID)
	}

	// Cycle detection — implemented fully in Task 3
	if err := d.checkCycle(blockerID, blockedID); err != nil {
		return err
	}

	_, err := d.q.Exec(
		`INSERT OR IGNORE INTO deps (blocker_id, blocked_id) VALUES (?, ?)`,
		blockerID, blockedID,
	)
	return err
}

func (d *DB) GetBlockers(taskID string) ([]Task, error) {
	rows, err := d.q.Query(
		`SELECT t.id, t.title, t.description, t.status, t.block_reason,
		        t.assignee, t.parent_id, t.created_at, t.updated_at
		 FROM tasks t JOIN deps d ON t.id = d.blocker_id
		 WHERE d.blocked_id = ?`, taskID,
	)
	if err != nil {
		return nil, err
	}
	defer rows.Close()
	return scanTasks(rows)
}

func (d *DB) checkCycle(blockerID, blockedID string) error {
	// DFS from blockedID to see if blockerID is reachable
	visited := make(map[string]bool)
	return d.dfs(blockerID, blockedID, visited)
}

func (d *DB) dfs(target, current string, visited map[string]bool) error {
	if current == target {
		return fmt.Errorf("dependency would create a cycle")
	}
	if visited[current] {
		return nil
	}
	visited[current] = true

	rows, err := d.q.Query(
		`SELECT blocker_id FROM deps WHERE blocked_id = ?`, current,
	)
	if err != nil {
		return err
	}
	defer rows.Close()

	for rows.Next() {
		var id string
		if err := rows.Scan(&id); err != nil {
			return err
		}
		if err := d.dfs(target, id, visited); err != nil {
			return err
		}
	}
	return rows.Err()
}

func scanTasks(rows *sql.Rows) ([]Task, error) {
	var tasks []Task
	for rows.Next() {
		var t Task
		if err := rows.Scan(&t.ID, &t.Title, &t.Description, &t.Status,
			&t.BlockReason, &t.Assignee, &t.ParentID, &t.CreatedAt, &t.UpdatedAt); err != nil {
			return nil, err
		}
		tasks = append(tasks, t)
	}
	return tasks, rows.Err()
}

package commands

import (
	"encoding/json"
	"fmt"
	"os"
	"text/tabwriter"

	"github.com/dispatch-ai/dispatch/internal/db"
	"github.com/spf13/cobra"

func jsonFlag(cmd *cobra.Command) bool {
	j, _ := cmd.Flags().GetBool("json")
	return j
}

func printJSON(v any) {
	enc := json.NewEncoder(os.Stdout)
	enc.SetIndent("", "  ")
	enc.Encode(v)
}

func printTask(t *db.Task) {
	w := tabwriter.NewWriter(os.Stdout, 0, 4, 2, ' ', 0)
	fmt.Fprintf(w, "ID:\t%s\n", t.ID)
	fmt.Fprintf(w, "Title:\t%s\n", t.Title)
	fmt.Fprintf(w, "Status:\t%s\n", t.Status)
	if t.Description != "" {
		fmt.Fprintf(w, "Description:\t%s\n", t.Description)
	}
	if t.Assignee != nil {
		fmt.Fprintf(w, "Assignee:\t%s\n", *t.Assignee)
	}
	if t.BlockReason != nil {
		fmt.Fprintf(w, "Block Reason:\t%s\n", *t.BlockReason)
	}
	if t.ParentID != nil {
		fmt.Fprintf(w, "Parent:\t%s\n", *t.ParentID)
	}
	fmt.Fprintf(w, "Created:\t%s\n", t.CreatedAt)
	fmt.Fprintf(w, "Updated:\t%s\n", t.UpdatedAt)
	w.Flush()
}

func printTaskList(tasks []db.Task) {
	w := tabwriter.NewWriter(os.Stdout, 0, 4, 2, ' ', 0)
	fmt.Fprintf(w, "ID\tSTATUS\tTITLE\n")
	for _, t := range tasks {
		fmt.Fprintf(w, "%s\t%s\t%s\n", t.ID, t.Status, t.Title)
	}
	w.Flush()
}

func exitError(cmd *cobra.Command, err error) {
	if jsonFlag(cmd) {
		printJSON(map[string]string{"error": err.Error()})
	} else {
		fmt.Fprintln(os.Stderr, "error:", err)
	}
	os.Exit(1)
}

func openDB(cmd *cobra.Command) *db.DB {
	path, _ := cmd.Flags().GetString("db")
	database, err := db.Open(path)
	if err != nil {
		exitError(cmd, err)
	}
	return database
}

package db

type Note struct {
	ID        int     `json:"id"`
	TaskID    string  `json:"task_id"`
	Content   string  `json:"content"`
	Author    *string `json:"author"`
	CreatedAt string  `json:"created_at"`
}

func (d *DB) AddNote(taskID, content string, author *string) (*Note, error) {
	if _, err := d.GetTask(taskID); err != nil {
		return nil, err
	}
	result, err := d.q.Exec(
		`INSERT INTO notes (task_id, content, author) VALUES (?, ?, ?)`,
		taskID, content, author,
	)
	if err != nil {
		return nil, err
	}
	noteID, _ := result.LastInsertId()
	return d.getNote(int(noteID))
}

func (d *DB) getNote(id int) (*Note, error) {
	n := &Note{}
	err := d.q.QueryRow(
		`SELECT id, task_id, content, author, created_at FROM notes WHERE id = ?`, id,
	).Scan(&n.ID, &n.TaskID, &n.Content, &n.Author, &n.CreatedAt)
	if err != nil {
		return nil, err
	}
	return n, nil
}

func (d *DB) GetNotes(taskID string) ([]Note, error) {
	rows, err := d.q.Query(
		`SELECT id, task_id, content, author, created_at
		 FROM notes WHERE task_id = ? ORDER BY created_at ASC`, taskID,
	)
	if err != nil {
		return nil, err
	}
	defer rows.Close()
	var notes []Note
	for rows.Next() {
		var n Note
		if err := rows.Scan(&n.ID, &n.TaskID, &n.Content, &n.Author, &n.CreatedAt); err != nil {
			return nil, err
		}
		notes = append(notes, n)
	}
	return notes, rows.Err()
}

func (d *DB) ReadyTasks() ([]Task, error) {
	rows, err := d.q.Query(`
		SELECT t.id, t.title, t.description, t.status, t.block_reason,
		       t.assignee, t.parent_id, t.created_at, t.updated_at
		FROM tasks t
		WHERE t.status = 'open'
		  AND t.assignee IS NULL
		  AND NOT EXISTS (
		    SELECT 1 FROM deps d
		    JOIN tasks blocker ON d.blocker_id = blocker.id
		    WHERE d.blocked_id = t.id
		    AND blocker.status != 'done'
		  )
		ORDER BY (
		    SELECT COUNT(*) FROM deps d2
		    WHERE d2.blocker_id = t.id
		    AND EXISTS (SELECT 1 FROM tasks t2 WHERE t2.id = d2.blocked_id AND t2.status != 'done')
		  ) DESC,
		  t.created_at ASC
	`)
	if err != nil {
		return nil, err
	}
	defer rows.Close()
	return scanTasks(rows)
}

func (d *DB) ListTasks(status string, all bool) ([]Task, error) {
	query := `SELECT id, title, description, status, block_reason,
	                 assignee, parent_id, created_at, updated_at FROM tasks`
	var args []any
	var conditions []string

	if status != "" {
		conditions = append(conditions, "status = ?")
		args = append(args, status)
	} else if !all {
		conditions = append(conditions, "status != 'done'")
	}

	if len(conditions) > 0 {
		query += " WHERE " + conditions[0]
		for _, c := range conditions[1:] {
			query += " AND " + c
		}
	}
	query += " ORDER BY created_at ASC"

	rows, err := d.q.Query(query, args...)
	if err != nil {
		return nil, err
	}
	defer rows.Close()
	return scanTasks(rows)
}

func printTaskTree(tasks []db.Task) {
	// Build parent -> children map
	children := make(map[string][]db.Task) // parentID -> tasks
	var roots []db.Task
	for _, t := range tasks {
		if t.ParentID != nil {
			children[*t.ParentID] = append(children[*t.ParentID], t)
		} else {
			roots = append(roots, t)
		}
	}

	w := tabwriter.NewWriter(os.Stdout, 0, 4, 2, ' ', 0)
	fmt.Fprintf(w, "ID\tSTATUS\tTITLE\n")
	for _, root := range roots {
		printTreeNode(w, root, children, 0)
	}
	w.Flush()
}

func printTreeNode(w *tabwriter.Writer, t db.Task, children map[string][]db.Task, depth int) {
	indent := ""
	for i := 0; i < depth; i++ {
		indent += "  "
	}
	fmt.Fprintf(w, "%s%s\t%s\t%s%s\n", indent, t.ID, t.Status, indent, t.Title)
	for _, child := range children[t.ID] {
		printTreeNode(w, child, children, depth+1)
	}
}

func NewBatchCmd() *cobra.Command {
	return &cobra.Command{
		Use:   "batch",
		Short: "Execute commands from stdin in a single transaction",
		Args:  cobra.NoArgs,
		Run: func(cmd *cobra.Command, args []string) {
			database := openDB(cmd)
			defer database.Close()

			tx, err := database.BeginTx()
			if err != nil {
				exitError(cmd, err)
			}

			scanner := bufio.NewScanner(os.Stdin)
			lineNum := 0
			for scanner.Scan() {
				lineNum++
				line := strings.TrimSpace(scanner.Text())
				if line == "" || strings.HasPrefix(line, "#") {
					continue
				}
				if err := executeLine(tx, line); err != nil {
					tx.Rollback()
					exitError(cmd, fmt.Errorf("line %d: %w", lineNum, err))
				}
			}
			if err := scanner.Err(); err != nil {
				tx.Rollback()
				exitError(cmd, err)
			}
			if err := tx.Commit(); err != nil {
				exitError(cmd, err)
			}
			if jsonFlag(cmd) {
				printJSON(map[string]string{"status": "ok", "lines": fmt.Sprintf("%d", lineNum)})
			}
		},
	}
}

// splitArgs splits a command line respecting quoted strings.
func splitArgs(line string) []string {
	var args []string
	var current strings.Builder
	inQuote := false
	quoteChar := byte(0)

	for i := 0; i < len(line); i++ {
		c := line[i]
		if inQuote {
			if c == quoteChar {
				inQuote = false
			} else {
				current.WriteByte(c)
			}
		} else if c == '"' || c == '\'' {
			inQuote = true
			quoteChar = c
		} else if c == ' ' || c == '\t' {
			if current.Len() > 0 {
				args = append(args, current.String())
				current.Reset()
			}
		} else {
			current.WriteByte(c)
		}
	}
	if current.Len() > 0 {
		args = append(args, current.String())
	}
	return args
}

func executeLine(database *db.DB, line string) error {
	parts := splitArgs(line)
	if len(parts) == 0 {
		return nil
	}

	switch parts[0] {
	case "add":
		return batchAdd(db, parts[1:])
	case "edit":
		return batchEdit(db, parts[1:])
	case "dep":
		if len(parts) != 3 {
